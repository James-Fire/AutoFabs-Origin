can_use_autofab = {
	has_origin = origin_autofab
}

can_expand_factory = {
	free_district_slots > 0
	has_deposit = autofab
	has_planet_flag = has_autofab
	can_queue_factory = yes
}

can_upgrade_autofab = {
	has_deposit = autofab
	has_planet_flag = has_autofab
}

can_queue_factory = {
	NOR = {
		has_planet_flag = factory_expanding
		has_planet_flag = autofab_automation
	}
}

has_supported_district_set = {
	uses_district_set = standard
	uses_district_set = habitat
	uses_district_set = city_world
	uses_district_set = hive_world
	uses_district_set = machine_world
	uses_district_set = ring_world
}

has_energy_available = {
	OR = {
		AND = {	
			uses_district_set = standard
			num_free_districts = { type = district_generator value > 0 }
		}
		AND = {
			uses_district_set = hive_world
			uses_district_set = machine_world
			free_district_slots > 0
		}
		AND = {
			uses_district_set = habitat
			has_planet_flag = energy_habitat
			free_district_slots > 0
		}
	}
}
has_minerals_available = {
	OR = {
		AND = {	
			uses_district_set = standard
			num_free_districts = { type = district_mining value > 0 }
		}
		AND = {
			uses_district_set = hive_world
			uses_district_set = machine_world
			free_district_slots > 0
		}
		AND = {
			uses_district_set = habitat
			has_planet_flag = mining_habitat
			free_district_slots > 0
		}
	}
}
has_food_available = {
	OR = {
		AND = {	
			uses_district_set = standard
			num_free_districts = { type = district_farming value > 0 }
		}
		AND = {
			uses_district_set = hive_world
			free_district_slots > 0
		}
	}
}
