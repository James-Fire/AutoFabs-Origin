#Planet Scope
automation_active = {
	OR = {
		owner = {
			is_ai = yes
		}
		has_planet_flag = autofab_automation
		AND = {
			owner = {
				has_country_flag = autofab_automation
			}
			NOT = {
				has_planet_flag = autofab_automation_off
			}
		}
	}
}

can_afford_district = {
	OR = {
		AND = {
			uses_district_set = standard
			owner = {
				resource_stockpile_compare = {
					resource = minerals
					value >= @districtcost
				}
				resource_stockpile_compare = {
					resource = alloys
					value >= @factoryalloycost
				}
			}
		}
		AND = {
			uses_district_set = habitat
			owner = {
				resource_stockpile_compare = {
					resource = minerals
					value >= @habdistrictcost
				}
				resource_stockpile_compare = {
					resource = alloys
					value >= @factoryalloycost
				}
			}
		}
		AND = {
			uses_district_set = city_world
			owner = {
				resource_stockpile_compare = {
					resource = minerals
					value >= @citydistrictcost
				}
				resource_stockpile_compare = {
					resource = alloys
					value >= @factoryalloycost
				}
			}
		}
		AND = {
			OR = {
				uses_district_set = hive_world
				uses_district_set = machine_world
			}
			owner = {
				resource_stockpile_compare = {
					resource = minerals
					value >= @specialdistrictcost
				}
				resource_stockpile_compare = {
					resource = alloys
					value >= @factoryalloycost
				}
			}
		}
		AND = {
			uses_district_set = ring_world
			owner = {
				resource_stockpile_compare = {
					resource = minerals
					value >= @ringdistrictcost
				}
				resource_stockpile_compare = {
					resource = alloys
					value >= @factoryalloycost
				}
			}
		}
	}
}
